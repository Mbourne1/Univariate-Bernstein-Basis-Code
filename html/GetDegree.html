function [degree_calc] = GetDegree(fx,gx,BOOL_Q,BOOL_DENOM)

Get degree of the approximate Greatest common divisor of fx and gx

fx : coefficients of polynomial f, expressed in Bernstein Basis
gx : coefficients of polynomail g, expressed in Bernstein Basis
BOOL_Q : Include the Binomial coefficients of the solution vector x =
[v ;-u] in the coefficient matrix $A_k$ for stabalisation

BOOL_DENOM : Inclusion of the denominators in the coefficient matrix $A_{k}$
BOOL_DENOM : Inclusion of the denominators in the coefficient matrix $A_{k}$

FontSize   = 15;
LineWidth  = 1.5;
MarkerSize = 10;

m = length(fx)-1;
n = length(gx)-1;
minimum_degree = min(m,n);

alpha_vector    =   zeros(1,minimum_degree);
theta_vector    =   zeros(1,minimum_degree);

Y = [];

% For each subresultant $$S_{k}$$
for k = 1:1:minimum_degree


    % Normalise the coefficients of F by the geometric mean of the
    % terms that contain its coefficients in S(f,g)Q. Repeat this
    % for the polynomial G.

     fx_n    =   fx/GM4(fx,n,k,BOOL_DENOM);
     gx_n    =   gx/GM4(gx,m,k,BOOL_DENOM);


    [alpha,theta]   =   OptimalAlphaAndTheta(fx_n,gx_n,k,BOOL_DENOM,BOOL_Q);
    alpha_vector(k) =   alpha;
    theta_vector(k) =   theta;

    % Calculate the coefficients of the modified Bernstein basis
    % polynomials F2 and G2. Multiply G2 by alpha.

    fw_n =  fx_n.*theta.^(0:1:m); %\bar{f}(\theta,\omega)
    gw_n =  gx_n.*theta.^(0:1:n); %\bar{g}(\theta,\omega)
    gw_n =  alpha.*gw_n;


    % Construct the kth subresultant matrix for the optimal
    % values of alpha and theta.
    s  =   Subresultant(fw_n,gw_n,k,BOOL_Q,BOOL_DENOM);

    % Using QR Decomposition of the sylvester matrix
    [~,R] = qr(s);

    R = abs(R);
    %Get number of rows in R1
    [R1_rows,~] = size(diag(R));
    %Obtain R1
    R1 = R(1:R1_rows,1:R1_rows);
    %Normalise R1
    R1 = R1./norm(R1);

    R1_sum_rows = sum(R1,2);
    R1_diag = abs(diag(R));

    %N = sqrt(sum(abs(R1_diag).^2,2));
    N = sqrt(sum(abs(R1).^2,2));
    %N = sort(N,'descend');
    N = N./norm(N);

    %Scatter Plot Data
    ks = k.*ones(size(N));
    ns = 1:1:size(N,1);
    X = [ks N ns'];
    Y = [Y; X];

    a1 = N(1);
    a2 = N(length(N)-1);
    a3 = N(length(N));
    delta(k) = log10(a2)-log10(a3);


    %Problem 1 - Get max|r_{i,i}| ./ min |r_{i,i}|
    p1(k) = max(R1_diag)./min(R1_diag);

    %Problem 2 - Get 1./ min|r_{ii}|
    p2(k) = 1./min(R1_diag);

    %Problem 3 - Get max| sum_{j} a_{i}{j})  | ./  min| sum_{j} a_{i}{j}|
    p3(k) = max(R1_sum_rows)./min(R1_sum_rows);

    %Problem 4 - Get 1 ./ min| sum_{j} a_{i}{j}|
    p4(k) = 1./min(R1_sum_rows);

    % Problem 6 - product of diagonal
    p6(k) = log10(a1/a3);

    % Problem 7 - gap between 2 last diag elements
    p7(k) = delta(k);

end

[~,degree_calc] = min(diff(p6));
display('                                    ');
display('Degree of AGCD of noisy polynomials:');
disp(degree_calc);
